	picture = GFX_evt_continental
	show_sound = event_mystic_reveal
	location = ROOT

	is_triggered_only = yes

	option = {
		name = EXCELLENT
		change_pc = pc_continental

		hidden_effect = {
			random_tile = {
				limit = {
					OR = {

			remove_modifier = "cofh_se_geoengineering_tectonics"
			remove_modifier = "cofh_se_geoengineering_weather"
		}
	}
}
