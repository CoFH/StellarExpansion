# Diplomatic Hub
building_cofh_se_diplomatic_hub = {
	icon = "building_cofh_se_diplomatic_hub"
	base_buildtime = 720
	planet_unique = yes

	cost = {
		minerals = 350
	}

	potential = {
		planet = {
			NOT = { is_planet_class = pc_habitat }
		}
	}

	required_resources = {
		energy = 5
	}

	produced_resources = {
		influence = 1
	}

	country_modifier = {
		max_influence = 100
	}

	planet_modifier = {
		pop_government_ethic_attraction = -0.15
	}

	prerequisites = {
		"tech_cofh_se_galactic_prominence"
	}

	allow = {
		AND = {
			custom_tooltip = {
				fail_text = "requires_building_capital_2"
				planet = {
					OR = {
						has_building = "building_capital_2"
						has_building = "building_capital_3"
					}
				}
			}
			custom_tooltip = {
				fail_text = "requires_pops_15"
				planet = {
					count_pops = {
						count > 14
						limit = {
							is_colony_pop = yes
						}
					}
				}
			}
		}
	}

	ai_special_building = yes

	ai_weight = {
		weight = 1000
	}
}

			}
			custom_tooltip = {

	prerequisites = {
		"tech_alien_life_studies"
	}

	ai_special_building = yes

	ai_weight = {
		weight = 1000
		modifier = {
			factor = 0
			from = {
				owner = {
					NOR = {
						has_ethic = ethic_xenophobe
						has_ethic = ethic_fanatic_xenophobe
					}
				}
				NOT = { has_building = building_cofh_se_xeno_food }
			}
		}
	}
}
