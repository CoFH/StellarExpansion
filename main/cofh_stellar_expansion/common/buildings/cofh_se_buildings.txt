# Diplomatic Hub

	icon = "building_cofh_se_diplomatic_hub"
	base_buildtime = 720
	planet_unique = yes

	cost = {
		minerals = 400
		influence = 25
	}

	required_resources = {
		energy = 3
	}

	produced_resources = {
		influence = 1
	}

	country_modifier = {
		max_influence = 100
	}

	planet_modifier = {
		pop_ethic_shift = 0.15
	}

	prerequisites = {
		"tech_cofh_se_galactic_prominence"
	}

	allow = {
		custom_tooltip = {
			text = "requires_building_capital_2"
			planet = {
				OR = {
					has_building = "building_capital_2"
					has_building = "building_capital_3"
				}
			}
		}
		custom_tooltip = {
			text = "requires_pops_15"
			planet = {
				count_pops = {
					count > 14
					limit = {
						is_colony_pop = yes
					}
				}
			}
		}
	}

	ai_special_building = yes

	ai_allow = {
		has_any_resource = no
	}

	ai_weight = {
		factor = 1000
		modifier = {
			factor = 0
			from = { sector_controlled = yes }
		}
	}
}
